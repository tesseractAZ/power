#include <tunables/global>

# v0.9.44 — AppArmor profile for the EcoFlow Panel add-on.
#
# Home Assistant Supervisor loads this file as the LSM profile for the
# container. Beyond bumping the security rating, it provides
# defense-in-depth: even if a Fastify endpoint is exploited (e.g., the
# v0.9.6 write-command framework or the v0.9.32 TTS debug endpoints),
# the attacker is confined to the file/network/capability set granted
# here.
#
# Permissive enough for our runtime:
#   - Node 22 + npm + s6-overlay init system
#   - Fastify HTTP + WebSocket + telnet (ports 8787, 2323)
#   - SQLite recorder.db + JSONL writes under /data
#   - Outbound HTTPS to api-a.ecoflow.com + Open-Meteo + NWS
#   - Inbound HTTP from HA Core + LAN (HomePods/Sonos fetching audio)
#   - Outbound to http://supervisor/* for HA Core + Supervisor APIs
#   - MA TTS proxy URL fetches from /api/tts_proxy/*
#
# Restrictive enough to deny:
#   - kernel module loading, mount, ptrace, raw IO, sysrq
#   - reading /etc/shadow, /root/**, /etc/ssh/**
#   - writing to /proc/sys/* or /sys/** (host kernel tunables)
#
# If the add-on misbehaves after this profile lands, the first thing to
# check is /var/log/audit/audit.log on the host (Pi) — denied operations
# log there with "type=AVC". Add corresponding rules below and rebuild.

profile addon_ecoflow_panel flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # ─── Capabilities ──────────────────────────────────────────────────
  # Allow binding to privileged ports (8787 and 2323 are unprivileged
  # but s6-overlay needs net_bind_service for some pre-start tasks),
  # plus setuid/setgid for the s6 user-switching dance.
  capability net_bind_service,
  capability setuid,
  capability setgid,
  capability chown,
  capability dac_override,
  capability dac_read_search,
  capability fowner,
  capability fsetid,
  capability kill,
  capability sys_chroot,

  # Explicitly DENY the dangerous capabilities. AppArmor's default is
  # "allow if not denied" — but these are worth calling out so any
  # future change to the profile doesn't silently enable them.
  deny capability sys_admin,
  deny capability sys_module,
  deny capability sys_rawio,
  deny capability sys_ptrace,
  deny capability sys_boot,
  deny capability sys_time,
  deny capability mac_admin,
  deny capability mac_override,

  # ─── File system ───────────────────────────────────────────────────
  # General read access to the container rootfs (app code, libs,
  # certificates, locale data). The Dockerfile is the source of truth
  # for what files exist; this profile follows its layout.
  #
  # This blanket `file,` rule is INTENTIONALLY RETAINED (do not remove it).
  #
  # v1.7.0 removed it (security #3) in favour of a per-path allow list, but that
  # list had `/init ix,` (execute) while s6-overlay re-OPENS /init for READING
  # during shutdown/restart — the read was denied → `/bin/sh: can't open '/init':
  # Permission denied` crash-loop → add-on down. Reverted in v1.7.1.
  #
  # v1.7.2 tightens #3 the safe way instead: KEEP `file,` (so no read/exec can
  # ever be denied — the /init crash CANNOT recur) and layer WRITE-immutability
  # deny rules on the code/binary/library dirs in the deny section below (see the
  # `deny /app|/usr|/bin|/sbin|/lib|/lib64|/init … wl,` block). That closes the
  # dominant CWE-732 vector (persisting/tampering via WRITE) with zero read-denial
  # risk. STILL DEFERRED (needs a host audit.log boot-test, never inspection):
  # full READ-confinement (dropping `file,`) and write-locking the s6 init trees
  # (/command, /package) + /etc — their failure mode is an invisible init
  # crash-loop, so they can't be verified by the add-on-log EACCES method.
  file,
  signal,
  network,

  # s6-overlay init system (HA base image uses s6).
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/local/bin/** ix,
  /usr/sbin/** ix,
  /sbin/** ix,
  /lib/** mr,
  /lib/**.so* mr,
  /usr/lib/** mr,
  /etc/s6/** ix,
  /etc/s6-overlay/** ix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /etc/services.d/** rwix,
  /run/s6/** rwix,
  /run/s6-rc** rwix,
  /package/admin/s6/command/** ix,
  /command/** rwix,

  # Add-on source + node_modules. Read + execute (node calls dlopen
  # on native modules under node_modules/.../*.node, which need mix).
  /app/** r,
  /app/server/node_modules/**/*.node mix,
  /app/server/node_modules/**/*.so* mix,

  # Persistent storage (/data is volume-mounted by HA Supervisor).
  # SQLite needs the rwk permission for WAL + shm files.
  /data/** rwk,

  # Temp + runtime files.
  /tmp/** rwk,
  /var/tmp/** rwk,
  /run/** rwk,

  # Standard config files (read-only system info).
  /etc/passwd r,
  /etc/group r,
  /etc/hosts r,
  /etc/host.conf r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/services r,
  /etc/protocols r,
  /etc/localtime r,
  /etc/timezone r,
  /etc/ssl/** r,
  /etc/ca-certificates/** r,
  /etc/os-release r,
  /usr/share/ca-certificates/** r,

  # /proc — Node + s6 need to read process info. Writes to /proc/sys
  # are denied below.
  /proc/[0-9]*/** r,
  /proc/sys/kernel/random/uuid r,
  /proc/sys/kernel/ngroups_max r,
  /proc/cpuinfo r,
  /proc/meminfo r,
  /proc/loadavg r,
  /proc/uptime r,
  /proc/stat r,
  /proc/version r,
  /proc/self/** rw,
  /proc/[0-9]*/comm rw,

  # /sys — read-only access to network info (for our diagnostic
  # endpoints that report interface state).
  /sys/class/net/** r,
  /sys/devices/** r,

  # /dev — Node + crypto need urandom; tty for logging.
  /dev/null rw,
  /dev/zero r,
  /dev/random r,
  /dev/urandom r,
  /dev/tty rw,
  /dev/pts/** rw,
  /dev/stdin r,
  /dev/stdout w,
  /dev/stderr w,

  # ─── Deny: dangerous file ops ──────────────────────────────────────
  # Never read secrets, never write to kernel tunables, never touch
  # the audit subsystem.
  deny /etc/shadow r,
  deny /etc/gshadow r,
  deny /root/** rw,
  deny /home/** rw,
  deny /etc/ssh/** r,
  deny /proc/sys/** w,
  deny /sys/** w,
  deny /proc/sysrq-trigger w,
  deny /sys/kernel/** w,
  deny /sys/firmware/** rw,

  # v1.7.2 (security #3, CWE-732) — code/binary/library WRITE-immutability.
  # The blanket `file,` rule above is RETAINED, so every READ and EXEC stays
  # granted (this is why the v1.7.0 `/init` read-denial crash-loop CANNOT recur —
  # v1.7.0 replaced `file,` with `/init ix,` and denied s6's READ of /init; we do
  # not touch reads/execs at all). These deny rules subtract `w`+`l` on the
  # immutable surfaces the add-on process could be coerced into tampering with:
  # `w` blocks create (O_CREAT), truncate/overwrite (O_TRUNC/O_WRONLY), O_APPEND,
  # rename-into and unlink; `l` blocks planting a hardlink under a code dir.
  # m/r/i/x/k are untouched, so .so/.node executable-mmap load + exec still work.
  # This shuts the dominant CWE-732 vector: a compromised Fastify/telnet/TTS
  # process persisting a restart-surviving backdoor. chmod/chown/utimes are
  # capability-mediated (chown/fowner/fsetid, granted above), NOT `w`-mediated, so
  # s6 fix-attrs is unaffected.
  #
  # ★★v1.9.1 — the mask is `wl`, NOT `wal`. The append bit `a` and write bit `w`
  # are MUTUALLY EXCLUSIVE in a single apparmor rule on the stricter parser HAOS
  # ships in kernel 6.18 (`w` already covers O_APPEND). v1.7.2-v1.9.0 shipped
  # `wal`, which parsed on the OLD apparmor feature set but the compiled result
  # was CACHED — a host reboot into the new feature set forced a fresh compile,
  # which rejected `wal` ("Conflict 'a' and 'w' perms are mutually exclusive")
  # → the profile failed to load → the container could not start (whole alarm
  # DOWN until hot-patched). NEVER combine `a` with `w`; validate any mask change
  # with `apparmor_parser -T -Q` (fresh, cache-skipping) before shipping.
  #
  # NOTE: /command + /package (s6 init trees, currently `rwix`) are intentionally
  # NOT denied here — a mistaken denial in those PID1 trees is an INVISIBLE init
  # crash-loop (the v1.7.0 symptom) that the add-on-log verification can't catch;
  # tightening them requires a host audit.log boot-test, deferred. /etc stays
  # writable (s6 needs /etc/{cont-init,services,cont-finish}.d) — also audit-driven.
  deny /app/{,**} wl,
  deny /usr/{,**} wl,
  deny /bin/{,**} wl,
  deny /sbin/{,**} wl,
  deny /lib/{,**} wl,
  deny /lib64/{,**} wl,
  deny /init wl,

  # Deny mount / umount / pivot_root (no container escape attempts).
  deny mount,
  deny umount,
  deny pivot_root,
  deny remount,

  # Deny ptrace (no peeking at other processes).
  deny ptrace,
}
